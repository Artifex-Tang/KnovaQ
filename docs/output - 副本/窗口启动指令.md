### 图片插入Word（python-docx）

在最终 docx 输出时，用 python-docx 将 PNG 图片插入对应章节：

```python
from docx import Document
from docx.shared import Inches, Pt, Cm
from docx.enum.text import WD_ALIGN_PARAGRAPH

doc = Document('模板.docx')

# 插入图片到指定位置（在某个段落后）
def insert_image_after_paragraph(doc, para, image_path, caption_text, width_inches=5.5):
    """在指定段落后插入图片+图注"""
    # 插入图片
    new_para = para._element.addnext(
        doc.add_paragraph()._element
    )
    from docx.oxml.ns import qn
    from lxml import etree
    # 创建图片段落
    pic_para = doc.add_paragraph()
    pic_para.alignment = WD_ALIGN_PARAGRAPH.CENTER
    run = pic_para.add_run()
    run.add_picture(image_path, width=Inches(width_inches))

    # 添加图注
    cap_para = doc.add_paragraph()
    cap_para.alignment = WD_ALIGN_PARAGRAPH.CENTER
    cap_run = cap_para.add_run(caption_text)
    cap_run.font.size = Pt(10)
    cap_run.font.bold = True

# 使用示例
# 找到"3.1.2 功能组成"标题段落后插入功能组成图
# insert_image_after_paragraph(doc, target_para, 'docs/output/images/功能组成图.png', '图 1 功能组成图')
```
